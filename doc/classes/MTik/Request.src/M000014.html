  def callback(sentence)
    case @callbacks.length
    when 0
      return nil
    when 1
      return @callbacks[0].call(self, sentence)
    else
      result = Array.new
      @callbacks.each do |cb|
        result.push(cb.call(self, sentence))
      end
      return result
    end
  end
